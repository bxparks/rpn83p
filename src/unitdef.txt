# This file defines the list of units supported by the UNIT conversion menu
# functions.
#
# To preserve backwards compatibility with Denominate numbers (values with
# units) stored in persistent memory (e.g. stack, registers, variables), this
# file must be treated as *append* only. Previously defined units must never
# be removed, modified, or reordered.
#
# Classes [
#   {list of classes}
# ]
#
# UnitTable [
#   {list of units}
# ]
#
# unitclass = 'UnitClass' {label}
# unit = 'Unit' {label} {name} {unitClass} {scale} {baseUnit}
#
# - {label} is the identifier used in the assembly language program
# - {name} is the user-visible name of the object
# - {scale} {baseUnit} is the size of the unit measured in {baseUnit}

UnitClasses [
  UnitClass Length
  UnitClass Area
  UnitClass Volume
]

Units [
  # Length
  Unit NanoMeter nm Length 1e-9 Meter
  Unit MicroMeter <Smu>m Length 1e-6 Meter
  Unit MilliMeter mm Length 1e-3 Meter
  Unit CentiMeter cm Length 1e-2 Meter
  Unit Meter m Length 1.0 Meter
  Unit KiloMeter km Length 1e3 Meter
  Unit MegaMeter Mm Length 1e6 Meter

  Unit Mil mil Length 0.0254e-3 Meter # exact
  Unit Inch in Length 0.0254 Meter # exact
  Unit Foot ft Length 0.3048 Meter # exact
  Unit Yard yd Length 0.9144 Meter # 3 feet
  Unit Fanthom fanthom Length 1.8288 Meter # 6 feet
  Unit Mile mi Length 1609.344 Meter # exact
  Unit NauticalMile nmi Length 1852 Meter # exact

  # Area
  Unit SqMeter m<Scaret>2 Area 1.0 SqMeter
  Unit SqFeet ft<Scaret>2 Area 0.09290304 SqMeter # 0.3048*0.3048
]

# be removed, modified, or reordered.
#
# Classes [
#   {list of classes}
# ]
#

#   {list of units}
# ]

# unitclass = 'UnitClass' {label}
# unit = 'Unit' {label} {name} {unitClass} {baseUnit} {scale}
#
# - 'label' is the identifier used in the assembly language program
# - 'name' is the user-visible name of the object
